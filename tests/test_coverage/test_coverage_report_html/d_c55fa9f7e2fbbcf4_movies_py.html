7def gridframes(frames, width, height, shape=None): 

8 ''' 

9 Arranges a set of frames into a grid layout with the specified 

10 pixel dimensions and shape. 

11 

12 :param list frames: A list of frames to include in the grid layout. 

13 :param int width: The width of the output grid image, in pixels. 

14 :param int height: The height of the output grid image, in pixels. 

15 :param optional shape: 

16 Either the number of frames to include, or the number of rows and columns 

17 in the output grid image layout. 

18 :return: A new image containing the arrangement of the frames in a grid. 

19 ''' 

20 if shape is None: 

21 shape = len(frames) 

22 if type(shape) not in [list,tuple]: 

23 shape = math.ceil(math.sqrt(shape)) 

24 shape = (shape, shape) 

25 

26 dsize = (height, width, 3) 

27 cellsize = (height // shape[0], width // shape[1],3) 

28 grid = np.zeros(dsize, dtype=np.uint8) 

29 for i in range(shape[0]): 

30 for j in range(shape[1]): 

31 k = i * shape[1] + j 

32 if k >= len(frames): 

33 continue 

34 frame = frames[k] 

35 i0 = i * cellsize[0] 

36 j0 = j * cellsize[1] 

37 i1 = i0 + cellsize[0] 

38 j1 = j0 + cellsize[1] 

39 grid[i0:i1, j0:j1] = cv2.resize(frame, (cellsize[1], cellsize[0])) 

40 return grid

46def groupframes(frames, n, fun): 

47 ''' 

48 Applies the specified function to each group of n-frames. 

49 

50 :param iterable frames: A sequence of frames to process. 

51 :param int n: The number of frames in each group. 

52 :param callable fun: The function used to process each group of frames. 

53 :return: An iterable returning the results of applying the function to each group. 

54 ''' 

55 i = 0 

56 group = [] 

57 for frame in frames: 

58 group.append(frame) 

59 if len(group) >= n: 

60 yield fun(group) 

61 group.clear() 

62 i = i + 1

64def triggerclip(data, events, before=pd.Timedelta(0), after=pd.Timedelta(0)): 

65 ''' 

66 Split video data around the specified sequence of event timestamps. 

67 

68 :param DataFrame data: 

69 A pandas DataFrame where each row specifies video acquisition path and frame number. 

70 :param iterable events: A sequence of timestamps to extract. 

71 :param Timedelta before: The left offset from each timestamp used to clip the data. 

72 :param Timedelta after: The right offset from each timestamp used to clip the data. 

73 :return: 

74 A pandas DataFrame containing the frames, clip and sequence numbers for each event timestamp. 

75 ''' 

76 if before is not pd.Timedelta: 

77 before = pd.Timedelta(before) 

78 if after is not pd.Timedelta: 

79 after = pd.Timedelta(after) 

80 if events is not pd.Index: 

81 events = events.index 

82 

83 clips = [] 

84 for i,index in enumerate(events): 

85 clip = data.loc[(index-before):(index+after)].copy() 

86 clip['frame_sequence'] = list(range(len(clip))) 

87 clip['clip_sequence'] = i 

88 clips.append(clip) 

89 return pd.concat(clips)

91def collatemovie(clipdata, fun): 

92 ''' 

93 Collates a set of video clips into a single movie using the specified aggregation function. 

94 

95 :param DataFrame clipdata: 

96 A pandas DataFrame where each row specifies video path, frame number, clip and sequence number. 

97 This DataFrame can be obtained from the output of the triggerclip function. 

98 :param callable fun: The aggregation function used to process the frames in each clip. 

99 :return: The sequence of processed frames representing the collated movie. 

100 ''' 

101 clipcount = len(clipdata.groupby('clip_sequence').frame_sequence.count()) 

102 allframes = aeon.videoframes(clipdata.sort_values(by=['frame_sequence', 'clip_sequence'])) 

103 return groupframes(allframes, clipcount, fun)
